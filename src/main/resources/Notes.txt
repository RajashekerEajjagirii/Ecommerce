# If we are creating roles as  ROLE_ADMIN and ROLE_USER then UserInfoDetailsService creates authorities like "ROLE_ADMIN"/"ROLE_USER"
 so use hasRole("ADMIN") in securityConfig and with @PreAuthorize("ADMIN").

# If we are creating roles as  ADMIN and USER then UserInfoDetailsService creates authorities like "ADMIN"/"USER"
  (not "ROLE_ADMIN"/"ROLE_USER"), so use hasAuthority("ADMIN")) in securityConfig and
  with @PreAuthorize("hasAuthority('ADMIN')") at controller or method level.
